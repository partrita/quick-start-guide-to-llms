def test_prompt_deepseek(
    prompt, suppress=False, model="deepseek-chat", retries=10, **kwargs
):
    """A simple function to take in a prompt and run it through a given non-chat model with retries."""

    for attempt in range(retries):
        try:
            chat_completion = deepseek_client.chat.completions.create(
                messages=[
                    {
                        "role": "user",
                        "content": prompt,
                    }
                ],
                model=model,
                **kwargs,
            )
            answer = chat_completion.choices[0].message.content
            if not suppress:
                print(
                    f"PROMPT:\n------\n{prompt}\n------\nRESPONSE (from {model})\n------\n{answer}"
                )
            return answer
        except Exception as e:
            wait_time = 2**attempt + random.uniform(0, 1)  # Exponential backoff
            print(
                f"Attempt {attempt + 1} failed: {e}. Retrying in {wait_time:.2f} seconds..."
            )
            time.sleep(wait_time)

    raise RuntimeError(f"Failed after {retries} retries.")

def extract_num(string):
    pattern = r"[\d,]+"
